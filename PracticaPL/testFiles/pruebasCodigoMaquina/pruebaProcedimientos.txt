int resultado =1;
proc controlador (int decision){
	int vect[4] = {103, 7, 13, 4};
	switch(decision){
		case 1{
		vect[1] = 6;
		}
		case 2{
		vect[2] = 14;
		}
		case 3{
		vect[3] = 5;
		}
		default {
			#El valor decision no deberia valer algo que no sea 1 , 2 o 3 .
		}
	}
	resultado = vect[0];
}
controlador(resultado);
